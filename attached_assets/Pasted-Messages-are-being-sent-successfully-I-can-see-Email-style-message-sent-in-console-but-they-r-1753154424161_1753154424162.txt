1. Connect to Stream channels to get messages:
javascript// In your component, watch for channel messages
useEffect(() => {
  if (!streamClient) return;

  // Get all channels for the current user
  const loadChannels = async () => {
    const filter = { 
      type: 'messaging', 
      members: { $in: [currentUser.streamId] } 
    };
    
    const channels = await streamClient.queryChannels(filter);
    
    // Extract messages from channels
    const allMessages = [];
    for (const channel of channels) {
      const messages = await channel.query({ messages: { limit: 100 } });
      allMessages.push(...messages.messages);
    }
    
    setMessages(allMessages);
  };

  loadChannels();
  
  // Listen for new messages
  streamClient.on('message.new', (event) => {
    setMessages(prev => [...prev, event.message]);
  });

}, [streamClient, currentUser]);
